class TestEmotionDetection(unittest.TestCase):
    def test_emotion_detector(self):

        de_joy = emotion_detector("I am glad this happened")
        self.assertEqual(de_joy['dominant_emotion'], "joy")

        de_anger = emotion_detector("I am really mad about this")
        self.assertEqual(de_anger['dominant_emotion'], "anger")

        de_disgust = emotion_detector("I am disgusted just hearing about this")
        self.assertEqual(de_disgust['dominant_emotion'], "disgust")

        de_sadness = emotion_detector("I am so sad about this")
        self.assertEqual(de_sadness['dominant_emotion'], "sadness")

        de_fear = emotion_detector("I am really afraid that this will happen")
        self.assertEqual(de_fear['dominant_emotion'], "fear")
       
        

unittest.main()
